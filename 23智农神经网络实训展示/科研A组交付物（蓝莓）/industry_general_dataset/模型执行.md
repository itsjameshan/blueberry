# 一、eval_test.py 独立测试脚本（完全匹配D2文档参数）
```python
from ultralytics import YOLO

# ====================== D2固定参数，禁止修改 ======================
DATA_YAML = "blueberry1.v4i.yolov11/data.yaml"
SEED = 42
IMGSZ = 640
CONF_THRESH = 0.25
SPLIT = "test"  # 仅独立测试集，不使用train/val
OUTPUT_DIR = "./test_result/"
# 模型权重路径（根据实际存放位置调整）
WEIGHT_V11 = "blueberry1.v4i.yolov11/runs/train/weights/best.pt"
WEIGHT_V12 = "blueberry1.v4i.yolov11/runs/train_v12/weights/best.pt"
# =================================================================

def run_test(model_path, model_name):
    print(f"===== 开始执行 {model_name} 独立test集测试 =====")
    model = YOLO(model_path)
    # 执行测试评估
    results = model.val(
        data=DATA_YAML,
        split=SPLIT,
        imgsz=IMGSZ,
        conf=CONF_THRESH,
        seed=SEED,
        save_json=True,    # 输出预测json，用于筛选失败案例
        save_plots=True,   # 保存预测可视化图
        save_txt=True,
        project=OUTPUT_DIR,
        name=model_name,
        exist_ok=True
    )
    # 打印核心指标
    print(f"{model_name} 全局指标：")
    print(f"mAP50: {results.box.map50:.4f}")
    print(f"mAP50-95: {results.box.map:.4f}")
    print(f"Precision: {results.box.p:.4f}")
    print(f"Recall: {results.box.r:.4f}")
    return results

if __name__ == "__main__":
    # 分别测试YOLOv11n、YOLOv12s
    res11 = run_test(WEIGHT_V11, "YOLOv11n")
    res12 = run_test(WEIGHT_V12, "YOLOv12s")
    print("===== 全部独立测试执行完成，结果保存在 ./test_result/ =====")
```
运行命令（cmd）：
```cmd
python eval_test.py > test_run_log.txt
```

1. 将 `eval_test.py` 放到 blueberry-master 根目录；
2. 确认训练好的 best.pt 权重路径和脚本内一致；
